function doGet(e) {
  try {
    const action = (e && e.parameter && e.parameter.action ? String(e.parameter.action) : '').toLowerCase();
    if (action === 'teammembers') {
      return getTeamMembersResponse();
    }
    if (action === 'alldata') {
      return getAllDataResponse();
    }
    if (action === 'masterroster') {
      return createJsonResponse(getMasterRosterData());
    }
    if (action === 'shiftblocks') {
      return createJsonResponse(getShiftBlocksData());
    }
    if (action === 'shifttypes') {
      return createJsonResponse(getShiftTypesData());
    }
    if (action === 'limitgroups') {
      return createJsonResponse(getLimitGroupsData());
    }
    if (action === 'activityhistory') {
      return createJsonResponse(getActivityHistoryData());
    }
    return getRequestsResponse();
  } catch (error) {
    return createJsonResponse({ result: 'error', message: error.message || error.toString() });
  }
}

function doPost(e) {
  try {
    const payload = parseRequestBody(e);
    const action = String(payload.action || '').toLowerCase();

    if (action === 'submit') {
      return handleSubmit(payload);
    }
    if (action === 'update') {
      return handleUpdate(payload);
    }
    if (action === 'delete') {
      return handleDelete(payload);
    }
    if (action === 'uploadmasterroster') {
      return handleUploadMasterRoster(payload);
    }
    if (action === 'updaterequestapproval') {
      return handleUpdateRequestApproval(payload);
    }
    if (action === 'addshiftblock') {
      return handleAddShiftBlock(payload);
    }
    if (action === 'deleteshiftblock') {
      return handleDeleteShiftBlock(payload);
    }
    if (action === 'addshifttype') {
      return handleAddShiftType(payload);
    }
    if (action === 'updateshifttype') {
      return handleUpdateShiftType(payload);
    }
    if (action === 'deleteshifttype') {
      return handleDeleteShiftType(payload);
    }
    if (action === 'reordershifttypes') {
      return handleReorderShiftTypes(payload);
    }
    if (action === 'addlimitgroup') {
      return handleAddLimitGroup(payload);
    }
    if (action === 'updatelimitgroup') {
      return handleUpdateLimitGroup(payload);
    }
    if (action === 'deletelimitgroup') {
      return handleDeleteLimitGroup(payload);
    }
    if (action === 'addactivity') {
      return handleAddActivity(payload);
    }
    if (action === 'deleteactivity') {
      return handleDeleteActivity(payload);
    }
    if (action === 'updatesetting') {
      return handleUpdateSetting(payload);
    }

    throw new Error('Unknown action: ' + action);
  } catch (error) {
    return createJsonResponse({ result: 'error', message: error.message || error.toString() });
  }
}

function getRequestsData() {
  const sheet = getSheetByName(REQUEST_SHEET_NAME);
  if (!sheet) {
    createSheetIfNotExist(REQUEST_SHEET_NAME, ['ID', 'Timestamp', 'Name', 'Date', 'Day', 'Request', 'Status', 'Comment', 'ApprovalStatus', 'SwapPartner', 'RequestType']);
    return [];
  }

  const values = sheet.getDataRange().getValues();
  if (values.length <= 1) {
    return [];
  }

  const headers = ensureHeaders(sheet, values[0]);
  const data = values.slice(1).map((row) => {
    const record = {};
    headers.forEach((header, index) => {
      record[header] = row[index];
    });
    return record;
  });

  return data;
}

function getTeamMembersData() {
  const sheet = getSheetByName(TEAM_MEMBERS_SHEET_NAME);
  if (!sheet) {
    createSheetIfNotExist(TEAM_MEMBERS_SHEET_NAME, ['MemberName']);
    return [];
  }

  const rows = sheet.getDataRange().getValues();
  if (rows.length <= 1) {
    return [];
  }

  const members = [];
  for (let i = 1; i < rows.length; i += 1) {
    const value = rows[i][0];
    if (!value) continue;
    const parsed = String(value).trim();
    if (!parsed) continue;
    members.push(parsed);
  }

  return Array.from(new Set(members));
}

function handleSubmit(data) {
  const sheet = getSheetByName(REQUEST_SHEET_NAME);
  if (!sheet) {
    throw new Error(`Sheet "${REQUEST_SHEET_NAME}" not found.`);
  }

  const range = sheet.getDataRange();
  const values = range.getValues();
  const headers = ensureHeaders(sheet, values[0] || []);
  const indexes = buildHeaderIndex(headers);
  const rows = values.slice(1);

  const name = String(data.name || '').trim();
  const date = data.date;
  const request = String(data.request || '').trim();
  const comment = data.comment ? String(data.comment).trim() : '';
  const requestType = data.requestType ? String(data.requestType).trim() : 'Leave';
  const swapPartner = data.swapPartner ? String(data.swapPartner).trim() : '';

  if (!name || !date || !request) {
    throw new Error('Missing required fields for submit.');
  }

  const day = data.day ? String(data.day) : getWeekdayName(date);
  const timestamp = new Date();
  const id = Date.now();

  rows.forEach((row, rowIndex) => {
    const matchesName = normalizeString(row[indexes.Name]) === normalizeString(name);
    const matchesDate = row[indexes.Date] === date;
    const isActive = normalizeString(row[indexes.Status]) === normalizeString(STATUS_ACTIVE);
    if (matchesName && matchesDate && isActive) {
      sheet.getRange(rowIndex + 2, indexes.Status + 1).setValue(STATUS_OLD);
    }
  });

  let approvalStatus = APPROVAL_PENDING_ADMIN;
  if (requestType.toLowerCase() === 'swap') {
    approvalStatus = APPROVAL_PENDING_PARTNER;
  }

  const record = buildRowFromHeaders(headers, {
    ID: id,
    Timestamp: timestamp,
    Name: name,
    Date: date,
    Day: day,
    Request: request,
    Status: STATUS_ACTIVE,
    Comment: comment,
    ApprovalStatus: approvalStatus,
    SwapPartner: swapPartner,
    RequestType: requestType
  });

  sheet.appendRow(record);
  return createJsonResponse({ result: 'success', id });
}

function handleUpdate(data) {
  const sheet = getSheetByName(REQUEST_SHEET_NAME);
  if (!sheet) {
    throw new Error(`Sheet "${REQUEST_SHEET_NAME}" not found.`);
  }

  const range = sheet.getDataRange();
  const values = range.getValues();
  const headers = ensureHeaders(sheet, values[0] || []);
  const indexes = buildHeaderIndex(headers);
  const rows = values.slice(1);

  const targetId = String(data.id || '').trim();
  if (!targetId) {
    throw new Error('Missing request ID for update.');
  }

  const name = String(data.name || '').trim();
  const date = data.date;
  const request = String(data.request || '').trim();
  const comment = data.comment ? String(data.comment).trim() : '';
  const requestType = data.requestType ? String(data.requestType).trim() : 'Leave';
  const swapPartner = data.swapPartner ? String(data.swapPartner).trim() : '';

  if (!name || !date || !request) {
    throw new Error('Missing required fields for update.');
  }

  const day = data.day ? String(data.day) : getWeekdayName(date);
  const timestamp = new Date();
  const newId = Date.now();
  let found = false;

  rows.forEach((row, rowIndex) => {
    const currentId = String(row[indexes.ID] || '').trim();
    if (currentId === targetId) {
      const status = normalizeString(row[indexes.Status]);
      if (status === normalizeString(STATUS_ACTIVE)) {
        sheet.getRange(rowIndex + 2, indexes.Status + 1).setValue(STATUS_OLD);
      }
      found = true;
    }
  });

  if (!found) {
    throw new Error('Request not found for update.');
  }

  let approvalStatus = APPROVAL_PENDING_ADMIN;
  if (requestType.toLowerCase() === 'swap') {
    approvalStatus = APPROVAL_PENDING_PARTNER;
  }

  const record = buildRowFromHeaders(headers, {
    ID: newId,
    Timestamp: timestamp,
    Name: name,
    Date: date,
    Day: day,
    Request: request,
    Status: STATUS_ACTIVE,
    Comment: comment,
    ApprovalStatus: approvalStatus,
    SwapPartner: swapPartner,
    RequestType: requestType
  });

  sheet.appendRow(record);
  return createJsonResponse({ result: 'success', id: newId, previousId: targetId });
}

function handleDelete(data) {
  const sheet = getSheetByName(REQUEST_SHEET_NAME);
  if (!sheet) {
    throw new Error(`Sheet "${REQUEST_SHEET_NAME}" not found.`);
  }

  const range = sheet.getDataRange();
  const values = range.getValues();
  const headers = ensureHeaders(sheet, values[0] || []);
  const indexes = buildHeaderIndex(headers);
  const rows = values.slice(1);

  const targetId = String(data.id || '').trim();
  if (!targetId) {
    throw new Error('Missing request ID for delete.');
  }

  let found = false;

  rows.forEach((row, rowIndex) => {
    const currentId = String(row[indexes.ID] || '').trim();
    if (currentId === targetId) {
      const status = normalizeString(row[indexes.Status]);
      if (status === normalizeString(STATUS_ACTIVE)) {
        sheet.getRange(rowIndex + 2, indexes.Status + 1).setValue(STATUS_CANCELLED);
      }
      found = true;
    }
  });

  if (!found) {
    throw new Error('Request not found or already inactive.');
  }

  return createJsonResponse({ result: 'success', id: targetId });
}

function handleUploadMasterRoster(data) {
  const sheet = createSheetIfNotExist(MASTER_ROSTER_SHEET_NAME, ['Name', 'Date', 'Shift']);
  
  const lastRow = sheet.getLastRow();
  if (lastRow > 1) {
    sheet.getRange(2, 1, lastRow - 1, sheet.getLastColumn()).clearContent();
  }

  const rows = data.rows;
  if (Array.isArray(rows) && rows.length > 0) {
    const values = rows.map(r => [
      String(r.name || '').trim(),
      String(r.date || '').trim(),
      String(r.shift || '').trim()
    ]);
    sheet.getRange(2, 1, values.length, 3).setValues(values);
  }

  return createJsonResponse({ result: 'success', count: rows ? rows.length : 0 });
}

function handleReorderShiftTypes(data) {
  const sheet = getSheetByName(SHIFT_TYPES_SHEET_NAME);
  if (!sheet) {
    throw new Error('ShiftTypes sheet not found.');
  }

  const ids = data.ids;
  if (!Array.isArray(ids) || ids.length === 0) {
    throw new Error('Missing or invalid ids array.');
  }

  // Read current rows (skip header)
  const allValues = sheet.getDataRange().getValues();
  const header = allValues[0];
  const dataRows = allValues.slice(1);

  // Build a map id -> row data
  const rowMap = {};
  dataRows.forEach(function(row) {
    rowMap[String(row[0]).trim()] = row;
  });

  // Build the new ordered rows based on the provided ids array
  const reordered = ids.map(function(id) {
    return rowMap[String(id).trim()];
  }).filter(Boolean);

  // Clear the sheet except header
  if (sheet.getLastRow() > 1) {
    sheet.getRange(2, 1, sheet.getLastRow() - 1, sheet.getLastColumn()).clearContent();
  }

  // Write reordered rows back
  reordered.forEach(function(row, i) {
    sheet.getRange(i + 2, 1, 1, row.length).setValues([row]);
  });

  return createJsonResponse({ result: 'success' });
}

function parseRequestBody(e) {
  if (!e || !e.postData || !e.postData.contents) {
    throw new Error('Missing request body.');
  }
  return JSON.parse(e.postData.contents);
}

function getSheetByName(name) {
  return SpreadsheetApp.getActiveSpreadsheet().getSheetByName(name);
}

function createJsonResponse(payload) {
  return ContentService.createTextOutput(JSON.stringify(payload))
    .setMimeType(ContentService.MimeType.JSON);
}

function buildHeaderIndex(headers) {
  return headers.reduce((acc, header, index) => {
    acc[header] = index;
    return acc;
  }, {});
}

function buildRowFromHeaders(headers, valuesByHeader) {
  const row = headers.map((header) => (header in valuesByHeader ? valuesByHeader[header] : ''));
  return row;
}

function createSheetIfNotExist(name, defaultHeaders) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(name);
  if (!sheet) {
    sheet = ss.insertSheet(name);
    sheet.getRange(1, 1, 1, defaultHeaders.length).setValues([defaultHeaders]);
  }
  return sheet;
}

function ensureHeaders(sheet, headers) {
  const required = ['ID', 'Timestamp', 'Name', 'Date', 'Day', 'Request', 'Status', 'Comment', 'ApprovalStatus', 'SwapPartner', 'RequestType'];
  let modified = false;
  const extendedHeaders = [...headers];
  
  required.forEach((req) => {
    if (extendedHeaders.indexOf(req) === -1) {
      extendedHeaders.push(req);
      modified = true;
    }
  });

  if (modified) {
    sheet.getRange(1, 1, 1, extendedHeaders.length).setValues([extendedHeaders]);
  }
  return extendedHeaders;
}

function normalizeString(value) {
  return String(value || '').trim().toLowerCase();
}

function getWeekdayName(dateValue) {
  if (!dateValue) return '';
  const date = new Date(dateValue);
  if (isNaN(date.getTime())) {
    return '';
  }
  return date.toLocaleDateString('en-US', { weekday: 'long' });
}

  const sheet = getSheetByName(ACTIVITY_HISTORY_SHEET_NAME);
  if (!sheet) {
    createSheetIfNotExist(ACTIVITY_HISTORY_SHEET_NAME, ['ID', 'Timestamp', 'CustomText', 'Name', 'RequestType', 'Request', 'SwapPartner', 'Date', 'ApprovalStatus', 'Comment', 'Status']);
    return [];
  }

  const values = sheet.getDataRange().getValues();
  if (values.length <= 1) {
    return [];
  }

  const headers = values[0];
  const statusColIndex = headers.indexOf('Status');

  const data = values.slice(1).map((row) => {
    const record = {};
    headers.forEach((header, index) => {
      record[header] = row[index];
    });
    return record;
  });

  // Only return Active entries; rows without a Status column are treated as Active (backward-compat)
  return data.filter((record) => {
    if (statusColIndex === -1) return true;
    const status = String(record.Status || '').trim().toLowerCase();
    return status !== 'archived';
  });
}

function handleAddActivity(data) {
  const sheet = createSheetIfNotExist(ACTIVITY_HISTORY_SHEET_NAME, ['ID', 'Timestamp', 'CustomText', 'Name', 'RequestType', 'Request', 'SwapPartner', 'Date', 'ApprovalStatus', 'Comment', 'Status']);
  
  const id = String(Date.now());
  const timestamp = new Date().toISOString();
  const customText = String(data.customText || '').trim();
  const name = String(data.name || '').trim();
  const requestType = String(data.requestType || '').trim();
  const request = String(data.request || '').trim();
  const swapPartner = String(data.swapPartner || '').trim();
  const date = String(data.date || '').trim();
  const approvalStatus = String(data.approvalStatus || '').trim() || 'Approved';
  const comment = String(data.comment || '').trim();

  sheet.appendRow([id, timestamp, customText, name, requestType, request, swapPartner, date, approvalStatus, comment, 'Active']);
  return createJsonResponse({ result: 'success', id });
}

function handleDeleteActivity(data) {
  const sheet = getSheetByName(ACTIVITY_HISTORY_SHEET_NAME);
  if (!sheet) {
    throw new Error('ActivityHistory sheet not found.');
  }

  const values = sheet.getDataRange().getValues();
  const headers = values[0] || [];
  const targetId = String(data.id || '').trim();

  if (!targetId) {
    throw new Error('Missing activity ID for delete.');
  }

  // Find or create the Status column
  let statusColIndex = headers.indexOf('Status');
  if (statusColIndex === -1) {
    // Status column doesn't exist yet — add it to the header row
    statusColIndex = headers.length;
    sheet.getRange(1, statusColIndex + 1).setValue('Status');
  }

  let found = false;
  for (let i = 1; i < values.length; i++) {
    const currentId = String(values[i][0] || '').trim();
    if (currentId === targetId) {
      // Soft delete: mark as Archived, never physically remove the row
      sheet.getRange(i + 1, statusColIndex + 1).setValue('Archived');
      found = true;
      break;
    }
  }

  if (!found) {
    throw new Error('Activity ID not found.');
  }

  return createJsonResponse({ result: 'success', id: targetId });
}

function getSettingsData() {
  const sheet = getSheetByName(SETTINGS_SHEET_NAME);
  if (!sheet) {
    createSheetIfNotExist(SETTINGS_SHEET_NAME, ['Key', 'Value']);
    sheet.appendRow(['monthly_request_limit', '10']);
    return { monthly_request_limit: 10 };
  }
  const values = sheet.getDataRange().getValues();
  const settings = {};
  for (let i = 1; i < values.length; i++) {
    const key = String(values[i][0] || '').trim();
    const val = String(values[i][1] || '').trim();
    if (key) {
      settings[key] = val;
    }
  }
  // Ensure default is there
  if (settings.monthly_request_limit === undefined) {
    settings.monthly_request_limit = '10';
  }
  return settings;
}

function handleUpdateSetting(data) {
  const key = String(data.key || '').trim();
  const value = String(data.value || '').trim();
  if (!key) {
    throw new Error('Missing setting key');
  }
  const sheet = getSheetByName(SETTINGS_SHEET_NAME);
  if (!sheet) {
    createSheetIfNotExist(SETTINGS_SHEET_NAME, ['Key', 'Value']);
  }
  const values = sheet.getDataRange().getValues();
  let rowIdx = -1;
  for (let i = 1; i < values.length; i++) {
    if (String(values[i][0] || '').trim() === key) {
      rowIdx = i + 1;
      break;
    }
  }
  if (rowIdx !== -1) {
    sheet.getRange(rowIdx, 2).setValue(value);
  } else {
    sheet.appendRow([key, value]);
  }
  return createJsonResponse({ result: 'success', key: key, value: value });
}
